# Условие
# Улитка ползет по вертикальному шесту высотой hh метров, 
# поднимаясь за день на aa метров, а за ночь спускаясь на bb метров. 
# На какой день улитка доползет до вершины шеста?
# 
# Программа получает на вход натуральные числа hh, aa, bb.
# 
# Программа должна вывести одно натуральное число. 
# Гарантируется, что a>ba>b.

from math import ceil

h = int(input())
a = int(input())
b = int(input())

print((h-b-1)//(a-b)+1)
