# Условие
# Дано натуральное число A. 
# Определите, каким по счету числом Фибоначчи оно является, 
# то есть выведите такое число n, что ?n = A. 
# Если А не является числом Фибоначчи, выведите число -1.

a = int(input())
if a == 0:
    print(0)
else:
    i, f1, f2 = 0, 0, 1
    while f1 < a:
        f1, f2 = f2, f1 + f2
        i += 1
        if f1 == a:
            print(i)
    if f1 > a:
        print(-1)
